# app_example/CMakeLists.txt — registers user sources into the SDK.
#
# Two ways to add user code:
#
# (A) PIO style: drop .c/.h files into ../src/. They will be auto-bridged
#     by platform-realtek-ameba via the OPTIONAL include() below. This keeps
#     the file list maintained automatically.
#
# (B) SDK style: list files explicitly here with ameba_list_append. Use
#     this when you need fine-grained control (per-file flags, conditional
#     sources, etc.).
#
# Both styles can coexist.

set(private_sources)
set(_pio_src_include_dirs)

# (B) SDK-style explicit list — at minimum we need the entry point.
ameba_list_append(private_sources app_main.c)

# (A) PIO-style: include the auto-generated fragment listing src/*.c and
# accumulating header dirs into _pio_src_include_dirs. Generated by
# platform-realtek-ameba on every `pio run`. OPTIONAL means the include
# silently no-ops if you've removed the src/ bridge.
include("${CMAKE_CURRENT_SOURCE_DIR}/_pio_src_fragment.cmake" OPTIONAL)

ameba_add_internal_library(app_example
    p_SOURCES
        ${private_sources}
)

# Apply the src/ include dirs (if any) AFTER the library has been created.
# c_CURRENT_TARGET_NAME is set by ameba_add_internal_library above (the
# function exports it via PARENT_SCOPE -- see SDK cmake/utility.cmake).
if(_pio_src_include_dirs)
    target_include_directories(${c_CURRENT_TARGET_NAME} PRIVATE ${_pio_src_include_dirs})
endif()
